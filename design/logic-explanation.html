// ─── ① UserDetailsServiceImpl.loadUserByUsername(email) ───
// Spring Security が POST /app/login 受信時に自動呼び出し

FUNCTION loadUserByUsername(email)

  // DB 検索（ソフトデリート除外）
  user = authMapper.findByEmail(email)
  // SQL: SELECT ... FROM users WHERE email=? AND deleted_at IS NULL

  IF user が null THEN
    THROW UsernameNotFoundException  // → /app/login?error=true
  END IF

  // ステータス確認
  statusActive = "ACTIVE".equalsIgnoreCase(user.status)

  // サブスクリプション確認（USER ロールのみ）
  IF user.roleId == 2 THEN
    subscriptionActive = authMapper.countEffectiveSubscriptions(user.id) > 0
    // SQL: WHERE status='ACTIVE' AND start_date<=TODAY AND (end_date IS NULL OR end_date>=TODAY)
  ELSE  // ADMIN / VET / STAFF は常に有効
    subscriptionActive = true
  END IF

  enabled = statusActive AND subscriptionActive

  RETURN LoginUser(id, roleId, name, email, passwordHash, enabled)

END FUNCTION

FUNCTION create(form, bindingResult, currentUser)

  // @Valid による自動バリデーション（petId 必須、scheduledAt 必須 など）
  IF bindingResult.hasErrors() THEN
    model.pets          = petService.list(...)        // フォーム再表示用
    model.availableSlots = generateAvailableSlots(...) // 空き枠を再取得
    RETURN "appointments/index"  // 入力画面に戻る（エラー表示）
  END IF

  // バリデーション通過 → Service に処理委譲
  appointmentService.createGeneralCare(form.petId, form.scheduledAt,
                                       form.note, form.channel, currentUser)
  ra.addFlashAttribute("success", "診療予約を登録しました")
  REDIRECT /app/appointments

END FUNCTION


// ─── ② Service 層：業務ロジックバリデーション ───
// AppointmentService.createGeneralCare

FUNCTION createGeneralCare(petId, scheduledAt, note, requestedChannel, currentUser)

  // プランアクセスチェック（USER はスタンダード以上が必要）
  IF NOT currentUser.isAdmin()
     AND NOT planAccessService.canUseAiSymptom(currentUser) THEN
    THROW BadRequestException("この機能はスタンダード以上で利用できます")
  END IF

  // 未来日時チェック
  IF scheduledAt == null OR scheduledAt <= LocalDateTime.now() THEN
    THROW BadRequestException("予約日時は未来日時を指定してください")
  END IF

  // 受付時間チェック（9:30〜17:00）
  IF scheduledAt.time < 09:30 OR scheduledAt.time >= 17:00 THEN
    THROW BadRequestException("診療受付時間外です（9:30〜17:00）")
  END IF

  // 重複チェック（同一日時に REQUESTED/CONFIRMED が存在）
  // SQL: WHERE deleted_at IS NULL AND scheduled_at=? AND status IN ('REQUESTED','CONFIRMED')
  IF appointmentMapper.countByScheduledAt(scheduledAt) > 0 THEN
    THROW BadRequestException("その時間帯はすでに予約済みです")
  END IF

  // ペット所有権チェック（VET/STAFF/ADMIN は全ペット参照可）
  pet = currentUser.hasStaffAccess()
        ? petMapper.findById(petId)
        : petMapper.findByIdAndOwnerUserId(petId, currentUser.id)
  IF pet == null THEN
    THROW NotFoundException("Pet not found")
  END IF
  IF pet.deceasedAt != null THEN
    THROW BadRequestException("永眠登録済みのペットはこの操作を利用できません")
  END IF

  // チャンネル決定（VISIT / ONLINE）
  channel = "VISIT"
  IF NOT currentUser.isAdmin() THEN
    IF planTier == STANDARD AND requestedChannel == "ONLINE" THEN
      THROW BadRequestException("スタンダードプランは来院予約のみ利用できます")
    ELSE IF planTier == PREMIUM AND requestedChannel == "ONLINE" THEN
      channel = "ONLINE"
    END IF
  ELSE IF requestedChannel == "ONLINE" THEN
    channel = "ONLINE"
  END IF

  // DB 登録（初期ステータス = "REQUESTED"）
  appointmentMapper.insert(AppointmentEntity(..., status="REQUESTED", channel=channel))

END FUNCTION

// ─── AppointmentService.listForApp ───────────────────────────────
// ロールに応じて取得範囲を切り替え、ページネーション済み結果を返す

FUNCTION listForApp(page, size, currentUser, sortBy)

  // ── 入力サニタイズ ──
  safePage = Math.max(page, 1)              // 1 以上に補正
  safeSize = Math.min(Math.max(size, 1), 100)  // 1〜100 に補正
  offset   = (safePage - 1) * safeSize
  sortByPet = "pet".equalsIgnoreCase(sortBy)

  IF currentUser.isAdmin() THEN
    // ADMIN：全ユーザーの予約を取得
    items = sortByPet
            ? appointmentMapper.findAllRowsOrderByPet(safeSize, offset)
            //   → ORDER BY p.name ASC, a.scheduled_at ASC, a.id ASC
            : appointmentMapper.findAllRows(safeSize, offset)
            //   → ORDER BY a.scheduled_at DESC, a.id DESC
    total = appointmentMapper.countAll()
    //   → SELECT COUNT(*) FROM appointments WHERE deleted_at IS NULL
    RETURN new PageResponse(items, safePage, safeSize, total)
  END IF

  // USER / VET / STAFF：owner_user_id で絞り込む
  items = sortByPet
          ? appointmentMapper.findRowsByOwnerUserIdOrderByPet(currentUser.id(), safeSize, offset)
          //   → AND a.owner_user_id = #{ownerUserId}  ORDER BY p.name ASC
          : appointmentMapper.findRowsByOwnerUserId(currentUser.id(), safeSize, offset)
          //   → AND a.owner_user_id = #{ownerUserId}  ORDER BY scheduled_at DESC
  total = appointmentMapper.countByOwnerUserId(currentUser.id())
  //   → SELECT COUNT(*) ... WHERE deleted_at IS NULL AND owner_user_id = #{ownerUserId}

  RETURN new PageResponse(items, safePage, safeSize, total)

END FUNCTION


// ─── SQL（MyBatis @Select — 実コード）────────────────────────────

// ① 全予約取得（ADMIN / 日時降順）findAllRows
SELECT a.id,
       p.name               AS "petName",
       u.name               AS "ownerName",
       a.appointment_type   AS "appointmentType",
       a.channel,
       a.scheduled_at       AS "scheduledAt",
       a.status, a.note,
       a.zoom_join_url      AS "zoomJoinUrl"
FROM appointments a
  INNER JOIN pets  p ON p.id = a.pet_id
  INNER JOIN users u ON u.id = a.owner_user_id
WHERE a.deleted_at IS NULL
ORDER BY a.scheduled_at DESC, a.id DESC
LIMIT #{limit} OFFSET #{offset}

// ② ペット名昇順（ADMIN）findAllRowsOrderByPet
//    SELECT … 同上  WHERE a.deleted_at IS NULL
ORDER BY p.name ASC, a.scheduled_at ASC, a.id ASC
LIMIT #{limit} OFFSET #{offset}

// ③ 個人絞り込み（USER 等 / 日時降順）findRowsByOwnerUserId
//    SELECT … 同上
WHERE a.deleted_at IS NULL
  AND a.owner_user_id = #{ownerUserId}   -- 自分の予約のみ
ORDER BY a.scheduled_at DESC, a.id DESC
LIMIT #{limit} OFFSET #{offset}

// ④ 個人絞り込み（USER 等 / ペット名昇順）findRowsByOwnerUserIdOrderByPet
//    SELECT … 同上
WHERE a.deleted_at IS NULL
  AND a.owner_user_id = #{ownerUserId}
ORDER BY p.name ASC, a.scheduled_at ASC, a.id ASC
LIMIT #{limit} OFFSET #{offset}

// ⑤ 重複チェック（予約作成前）countByScheduledAt
SELECT COUNT(*) FROM appointments
WHERE deleted_at IS NULL
  AND scheduled_at = #{scheduledAt}
  AND status IN ('REQUESTED', 'CONFIRMED')  -- 取消済みは除外

// getAuthorities(): Spring Security が内部で使用するロール文字列を返す
FUNCTION getAuthorities(roleId)
  RETURN SWITCH roleId
    1 → "ROLE_ADMIN"
    3 → "ROLE_VET"
    4 → "ROLE_STAFF"
    default → "ROLE_USER"
END FUNCTION

// ─── 承認（REQUESTED → CONFIRMED）VET / STAFF が実行 ─────────────────────

FUNCTION approve(id, actorUserId)

  existing = appointmentMapper.findById(id)
  IF existing == null THEN
    THROW NotFoundException("Appointment not found: " + id)
  END IF

  IF existing.status() != "REQUESTED" THEN
    THROW BadRequestException("申請中の予約のみ承認できます")
  END IF

  // ONLINE 予約なら Zoom ミーティング URL を発行（失敗時はフォールバック URL）
  zoomJoinUrl = existing.zoomJoinUrl()
  IF existing.channel() == "ONLINE" THEN
    zoomResult = zoomLinkService.createMeetingOrFallback(
                   existing.scheduledAt(), "Pet Life Plus Premium Online Care")
    zoomJoinUrl = zoomResult.joinUrl()   // ZoomMeetingResult はレコード型
  END IF

  // 全フィールドをコピーしたうえで status・zoomJoinUrl だけ差し替えた新エンティティを作成
  updated = NEW AppointmentEntity(
              existing.id(), existing.petId(), existing.ownerUserId(), existing.staffUserId(),
              existing.appointmentType(), existing.channel(), existing.scheduledAt(),
              "CONFIRMED", zoomJoinUrl,
              existing.note(), existing.slotId(),
              existing.deletedAt(), existing.createdAt(), existing.updatedAt())

  appointmentMapper.update(updated)   // UPDATE appointments SET … WHERE id=#{id}

  // 更新後エンティティを渡して予約者へ通知（承認時は Zoom リンクも本文に含む）
  notifyOwnerStatusChanged(updated, approved=true, actorUserId)
  // → notifications + notification_recipients テーブルへ INSERT、type="INFO"
  // → title: "診療予約が承認されました"

  RETURN toResponse(updated)

END FUNCTION


// ─── 却下（REQUESTED → CANCELED）VET / STAFF が実行 ──────────────────────

FUNCTION reject(id, actorUserId)

  existing = appointmentMapper.findById(id)
  IF existing == null THEN
    THROW NotFoundException("Appointment not found: " + id)
  END IF

  IF existing.status() != "REQUESTED" THEN
    THROW BadRequestException("申請中の予約のみ却下できます")
  END IF

  // 通知に渡すため CANCELED 状態のエンティティを先に構築（DB 更新は updateStatus のみ）
  canceled = NEW AppointmentEntity(
               existing.id(), existing.petId(), existing.ownerUserId(), existing.staffUserId(),
               existing.appointmentType(), existing.channel(), existing.scheduledAt(),
               "CANCELED", existing.zoomJoinUrl(),
               existing.note(), existing.slotId(),
               existing.deletedAt(), existing.createdAt(), existing.updatedAt())

  appointmentMapper.updateStatus(id, "CANCELED")   // UPDATE appointments SET status=? WHERE id=?

  // canceled エンティティを渡して予約者へ却下通知
  notifyOwnerStatusChanged(canceled, approved=false, actorUserId)
  // → type="INFO"、title: "診療予約が却下されました"

  RETURN toResponse(canceled)

END FUNCTION


// ─── ユーザー自身によるキャンセル（REQUESTED → CANCELED）USER が実行 ───────

FUNCTION cancelRequestedByOwner(id, currentUser)

  existing = appointmentMapper.findById(id)
  IF existing == null THEN
    THROW NotFoundException("Appointment not found: " + id)
  END IF

  // 所有者確認（status チェックより先に行う）
  IF currentUser == null
     OR NOT existing.ownerUserId().equals(currentUser.id()) THEN
    THROW BadRequestException("自分の予約のみキャンセルできます")
  END IF

  // REQUESTED 状態のみキャンセル可（CONFIRMED 後は不可）
  IF existing.status() != "REQUESTED" THEN
    THROW BadRequestException("申請中の予約のみキャンセルできます")
  END IF

  appointmentMapper.updateStatus(id, "CANCELED")

  // ADMIN 全員へ ALERT 通知（予約者本人ではなく管理者へ）
  notifyAdminsCanceledByOwner(existing, currentUser.id())
  // → userMapper.findAdminUserIds() で全 ADMIN を取得してループ
  // → type="ALERT"、title: "予約がキャンセルされました"

  RETURN get(id)   // DB から最新状態を再取得して返す

END FUNCTION
